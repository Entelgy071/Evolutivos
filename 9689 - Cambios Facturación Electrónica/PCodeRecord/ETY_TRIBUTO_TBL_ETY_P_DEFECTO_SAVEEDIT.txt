/*Inicio 001*/
Local Rowset &rsETY_TRIBUTO_TBL;
Local string &P_DEFECTO;

&rsETY_TRIBUTO_TBL = GetLevel0().GetRow(1).GetRowset(Scroll.ETY_TRIBUTO_TBL);
For &i = 1 To &rsETY_TRIBUTO_TBL.ActiveRowCount
   &Cod = &rsETY_TRIBUTO_TBL.GetRow(&i).ETY_TRIBUTO_TBL.MATCHNAME.Value;
   If &rsETY_TRIBUTO_TBL.GetRow(&i).ETY_TRIBUTO_TBL.ETY_P_DEFECTO.Value = "Y" Then
      &P_DEFECTO = "Y";
   End-If;
End-For;

If &P_DEFECTO <> "Y" And
      All(&Cod) Then
   Error ("En la parametrización de Tributos debe seleccionar un campo Por Defecto. ");
End-If;

/*Fin 001*/
